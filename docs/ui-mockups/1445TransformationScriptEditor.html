Transformation Script (ES6)

/** * @param {object} data - The input JSON data. * @returns {object} The transformed JSON data. */ function transform(data) { if (!Array.isArray(data)) { return { 'error': 'Input is not an array' }; } return data.map(item => ({ id: item.user_id, fullName: `${item.first_name} ${item.last_name}`, isActive: item.status === 'active' })); }
